### dbt_profile.yml

```yaml
name: string

config-version: 2
version: version

profile: profilename

source-paths: [directorypath] - default model
data-paths: [directorypath] -default data
test-paths: [directorypath] - default tests
analysis-paths: [directorypath] - default analysis
macro-paths: [directorypath] - default macros90-
snapshot-paths: [directorypath]
docs-paths: [directorypath]
asset-paths: [directorypath]

target-path: directorypath
log-path: directorypath
modules-path: directorypath

clean-targets: [directorypath]

query-comment: string

require-dbt-version: version-range 

quoting:
  database: true | false
  schema: true | false
  identifier: true | false

models:
  <model-configs>

seeds:
  <seed-configs>

snapshots:
  <snapshot-configs>

sources:
  <source-configs>

vars:
  <variables>

on-run-start: sql-statement | [sql-statement]
on-run-end: sql-statement | [sql-statement]

name: 'datahub_refinery'
version: '1.0.0'
config-version: 2
```

#### Sample

```yaml

profile: 'endeavour-adb'
source-paths: ["models"]
analysis-paths: ["analysis"]
test-paths: ["tests"]
data-paths: ["data"]
macro-paths: ["macros"]
snapshot-paths: ["snapshots"]

target-path: "target"  # directory which will store compiled SQL files
clean-targets:         # directories to be removed by `dbt clean`
    - "target"
    - "dbt_modules"
    - "logs"

#### Configuring models
# Full documentation: https://docs.getdbt.com/docs/configuring-models
models:
  datahub_refinery:
      raw:
          materialized: view
```
